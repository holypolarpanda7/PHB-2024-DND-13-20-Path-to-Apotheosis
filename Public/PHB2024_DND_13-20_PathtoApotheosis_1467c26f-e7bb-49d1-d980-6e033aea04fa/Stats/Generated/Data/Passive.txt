// Signature Spells (Level 20): BG3 adaptation. PHB picks two 3rd-level spells to
// cast for free each rest; here, twice per Short Rest, casting any 3rd-level spell
// (at 3rd level) refunds the spell slot, so the player spends the free casts on the
// 3rd-level spells they treat as signatures. Gated by the SignatureSpellCharge
// resource (max 2, replenished on Short Rest). Mirrors the Expert Divination slot-
// refund pattern (OnCast + RestoreResource(SpellSlot)).
new entry "Wizard_SignatureSpells"
type "PassiveData"
data "DisplayName" "he2a3b4c500000000000000000000033"
data "Description" "he2a3b4c600000000000000000000034"
data "Icon" "PassiveFeature_ArcaneTradition"
data "Properties" "Highlighted"
data "StatsFunctorContext" "OnCast"
data "Conditions" "IsSpell() and not IsCantrip() and SpellPowerLevelEqualTo(3) and HasActionResource('SignatureSpellCharge',1,0,false)"
data "StatsFunctors" "RestoreResource(SELF,SpellSlot,1,3);UseActionResource(SELF,SignatureSpellCharge,1)"
